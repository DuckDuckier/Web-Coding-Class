Tabla de atributos de la etiqueta <input>

Atributo Descripción Ejemplo
type Define el tipo de entrada (text, number, email, etc.) <input type="text">
src Fuente de imagen para type="image" <input type="image" src="boton.png">
value Valor por defecto del campo <input value="Hola">
alt Texto alternativo para imágenes <input type="image" alt="Enviar">
height Altura de imagen (type="image") <input type="image" height="40">
width Ancho de imagen (type="image") <input type="image" width="100">
size Define el ancho del campo en caracteres <input type="text" size="30">
autocomplete Activa o desactiva el autocompletado <input autocomplete="off">
checked Marca una opción por defecto (checkbox/radio) <input type="checkbox" checked>
disabled Desactiva el campo <input disabled>
readonly Solo lectura, no se puede editar <input readonly value="Solo lectura">
name Nombre del campo para el formulario <input name="usuario">
max Valor máximo permitido <input type="number" max="100">
min Valor mínimo permitido <input type="number" min="10">
maxlength Número máximo de caracteres <input maxlength="20">
list Conecta con un datalist para sugerencias <input list="opciones"> <datalist id="opciones"> <option value="Opción 1"> <option value="Opción 2"> </datalist>
multiple Permite seleccionar múltiples archivos <input type="file" multiple>
step Incremento permitido en campos numéricos <input type="number" step="5">
formenctype Codificación del formulario al enviar <input type="submit" formenctype="multipart/form-data">
formtarget Destino donde se abre la respuesta <input type="submit" formtarget="_blank">
formaction URL a la que se envía el formulario <input type="submit" formaction="procesar.php">
formnovalidate Evita la validación del formulario <input type="submit" formnovalidate>
pattern Expresión regular para validar el formato <input pattern="[A-Za-z]+">
