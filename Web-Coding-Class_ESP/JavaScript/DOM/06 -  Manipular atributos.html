Manipulación de atributos

Accede, modifica y elimina atributos de elementos HTML con JavaScript

Ejemplo: Acceder, modificar y eliminar atributos

HTML + JavaScript para trabajar con atributos del DOM

<!DOCTYPE html>
<html>
<head>
  <title>Manipulación de atributos del DOM</title>
</head>
<body>
  <h1 id="titulo">Título</h1>
  <img src="imagen.jpg" alt="Imagen" id="imagen" width="200" height="200">

  <!-- Botones para ejecutar funciones -->
  <button onclick="mostrarAtributo()">Mostrar Atributo</button>
  <button onclick="cambiarAtributo()">Cambiar Atributo</button>
  <button onclick="verificarAtributo()">Verificar Atributo</button>
  <button onclick="eliminarAtributo()">Eliminar Atributo</button>

  <script>
    // Obtener y mostrar el valor del atributo 'src'
    function mostrarAtributo() {
      const imagen = document.getElementById('imagen');
      const src = imagen.getAttribute('src');
      console.log('Atributo src:', src);
    }

    // Cambiar los atributos 'src' y 'alt' de la imagen
    function cambiarAtributo() {
      const imagen = document.getElementById('imagen');
      imagen.setAttribute('src', 'nueva-imagen.jpg');
      imagen.setAttribute('alt', 'Nueva Imagen');
      console.log('Atributos cambiados');
    }

    // Verificar si la imagen tiene el atributo 'alt'
    function verificarAtributo() {
      const imagen = document.getElementById('imagen');
      const tieneAlt = imagen.hasAttribute('alt');
      console.log('¿La imagen tiene el atributo alt?', tieneAlt);
    }

    // Eliminar los atributos 'width' y 'height' de la imagen
    function eliminarAtributo() {
      const imagen = document.getElementById('imagen');
      imagen.removeAttribute('width');
      imagen.removeAttribute('height');
      console.log('Atributos eliminados');
    }
  </script>
</body>
</html>
Este ejemplo muestra cómo manipular atributos del DOM:
  • getAttribute: obtiene el valor de un atributo.
  • setAttribute: modifica o agrega un atributo.
  • hasAttribute: verifica si un atributo existe.
  • removeAttribute: elimina un atributo del elemento.
Estos métodos permiten actualizar dinámicamente imágenes, enlaces, textos alternativos y más. Son esenciales para crear interfaces interactivas y adaptables.
