Algoritmo de búsqueda

Encuentra un elemento dentro de una estructura de datos.


function buscarElemento(array, valor) {
  for (let i = 0; i < array.length; i++) {
    if (array[i] === valor) return i;
  }
  return -1;
}

let nombres = ["Ana", "Luis", "Joaquín"];
console.log(buscarElemento(nombres, "Joaquín")); // Devuelve 2

Busca el nombre "Joaquín" en el array y devuelve su posición.

Algoritmo de ordenamiento

Organiza elementos en un orden específico.


let numeros = [5, 2, 9, 1];
numeros.sort((a, b) => a - b);
console.log(numeros); // [1, 2, 5, 9]

Ordena los números de menor a mayor usando el método sort().

Algoritmo recursivo

Se llama a sí mismo para resolver subproblemas.


function factorial(n) {
  if (n === 0) return 1;
  return n * factorial(n - 1);
}

console.log(factorial(5)); // 120

Calcula el factorial de un número usando recursión.

Algoritmo de grafos

Trabaja con nodos y conexiones, como rutas o redes.


let grafo = {
  A: ["B", "C"],
  B: ["D"],
  C: ["D"],
  D: []
};

function recorridoDFS(nodo, visitados = []) {
  if (!visitados.includes(nodo)) {
    visitados.push(nodo);
    grafo[nodo].forEach(vecino => recorridoDFS(vecino, visitados));
  }
  return visitados;
}

console.log(recorridoDFS("A")); // ["A", "B", "D", "C"]

Recorre un grafo en profundidad desde el nodo "A".

Algoritmo de cifrado

Transforma datos para proteger su contenido.


function cifrar(texto) {
  let resultado = "";
  for (let i = 0; i < texto.length; i++) {
    resultado += String.fromCharCode(texto.charCodeAt(i) + 1);
  }
  return resultado;
}

console.log(cifrar("hola")); // "ipmb"

Cifra el texto desplazando cada letra una posición en el código ASCII.
